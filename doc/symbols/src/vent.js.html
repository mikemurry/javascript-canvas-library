 25 /**
 26  * @description Safely augments the VENT namespace to include the specified path.
 27  * @param ns {String} A dot separated namespace. Prefixing with "VENT." is optional.
 28  * @return {Object} Returns the VENT object.
 29  */
 30 
 31 VENT.namespace = function(ns) {
 32 
 33     var parts, parent, i, count;
 34 
 35     parent = VENT;
 36     parts = ns.split(".");
 37 
 38     // Skip Redundant Top Level
 39     if (parts[0] === "VENT") {
 40         parts = parts.slice(1);
 41     }
 42 
 43     count = parts.length;
 44 
 45     for (i=0; i < count; i++) {
 46 
 47         // Create a property if it doesn't exist.
 48         if (typeof parent[parts[i]] === "undefined") {
 49             parent[parts[i]] = {};
 50         }
 51 
 52         // Set parent to new property.
 53         parent = parent[parts[i]];
 54 
 55     }
 56 
 57     return parent;

 65     /**
 66      * @description Returns an integer between the specified minimum (inclusive) and maximum (exclusive).
 67      * @param min {Number} The minimum range for the returned integer (inclusive).
 68      * @param max {Number} The maximum range for the returned integer (exclusive).
 69      * @return {Number}

 71 
 72     randomInt: function randomInt(min, max) {
 73         if (min && max) {
 74             return Math.floor((max - min) * Math.random()) + min;
 75         } else { VENT.warn("utilities.randomInt() Error: Missing min or max argument."); }
 76         return false;
 77     },
 78 
 79     /**
 80      * @description Calculates a point {x,y} at the specified angle and distance from the base point.
 81      * @param center {Object} The center point {x,y} to base the resulting point.
 82      * @param angle {Number} The number of degrees to rotate the resulting point around the center (0 is pointing right).
 83      * @param distance {Number} The number of pixels between the center and the resulting point.
 84      * @return {Object|Undefined} The resulting point expressed as an object with 'x' and 'y' properties.
 85      */
 86 
 87     pointTangent: function pointTangent(center, angle, distance) {
 88         if (center && angle !== null && distance !== null) {
 89             if (this.isPoint(center)) {
 90                 var radians = this.radians;
 91                 return { x: (distance * Math.cos(radians(angle))) + center.x, y: (distance * Math.sin(radians(angle))) + center.y };
 92             } else { VENT.warn("Could not calculate tangent. Invalid point data. (X: " + center.x + ", Y: " + center.y + ")"); }
 93         } else { VENT.warn("Could not calculate tangent. Missing required data. (Point: " + center + ", Angle: " + angle + ", Distance: " + distance + ")"); }
 94         return undefined;
 95     },
 96 
 97     /**
 98      * @description Interpolates between two points {x,y}.
 99      * @param a {Object} The first point {x,y}.
100      * @param b {Object} The second point {x,y}.
101      * @param ratio {Number} The amount to interpolate between the two points. Between 0 and 1.
102      * @return {Object|Undefined} The interpolated point {x,y}.
103      */
104 
105     pointLerp: function pointLerp(a,b,ratio) {
106         if (!ratio || ratio < 0) { ratio = 0; }
107         else if (ratio > 1) { ratio = 1; }
108         if (a && b && ratio !== null) {
109             if (this.isPoint(a) && this.isPoint(b)) {
110                 return { x: a.x + ((b.x - a.x) * ratio), y: a.y + ((b.y - a.y) * ratio) };
111             } else { VENT.warn("Could not interpolate point. Invalid point."); }
112         } else { VENT.warn("Could not interpolate point. Missing required data."); }
113         return undefined;
114     },
115 
116     /**
117      * @description Converts degrees into radians.
118      * @param degrees {Number}
119      * @return {Number}
120      */
121 
122     radians: function radians(degrees) {
123         return (Math.PI / 180) * degrees;
124     },
125 
126     /**
127      * @description Validates if the argument is a point, an object with 'x' and 'y' properties.
128      * @param point {Object} The object to test.
129      * @return {Boolean}
130      */
131 
132     isPoint: function isPoint(point) {
133         return (point.hasOwnProperty("x") && point.hasOwnProperty("y"));
134     },

136     createFragment: function createFragment(id, htmlStr) {
137         var frag = document.createDocumentFragment(), temp = document.createElement("div");
138         temp.id = id;
139         temp.innerHTML = htmlStr;
140         while (temp.firstChild) { frag.appendChild(temp.firstChild); }
141         return frag;
142     }

195     /**
196      * @description Draws a rectangle on the canvas.
197      * @param ctx {Object} The canvas context to draw on.
198      * @param x {Number} The distance from the left edge of the canvas to draw the left edge of the rectangle.
199      * @param y {Number} The distance form the top edge of the canvas to draw the top edge of the rectangle.
200      * @param w {Number} The width, in pixels, of the rectangle.
201      * @param h {Number} The height, in pixels, of the rectangle.
202      * @param fillStyle {String} The canvas style used to fill the rectangle.
203      * @param strokeStyle {String} The canvas style used to stroke the rectangle.
204      * @param lineWidth {Number} The stroke width of the rectangle.
205      * @private
206      */
207 
208     function _drawRectangle(ctx, x, y, w, h, fillStyle, strokeStyle, lineWidth) {
209 
210         if (fillStyle) {
211             ctx.fillStyle = fillStyle;
212             ctx.fillRect(x, y, w, h);
213         }
214 
215         if (strokeStyle) {
216             ctx.strokeStyle = strokeStyle;
217             ctx.lineWidth = lineWidth || 1;
218             ctx.strokeRect(x, y, w, h);
219         }
220 
221     }
222 
223     /**
224      * @description Draws a path on the canvas.
225      * @param ctx {Object} The canvas context used to draw the path.
226      * @param steps {Object[]} An array of points {x,y} to draw sequentially.
227      * @param strokeStyle {String} The canvas style used to stroke the path.
228      * @param lineWidth {Number} The width of the path.
229      * @private
230      */
231 
232     function _drawPath(ctx, steps, strokeStyle, lineWidth) {
233 
234         var length = steps.length, i;
235 
236         ctx.strokeStyle = strokeStyle || "#000000";
237         ctx.lineWidth = lineWidth || 1;
238         ctx.beginPath();
239         ctx.moveTo(steps[0].x, steps[0].y);
240         for (i = 1; i<length; i++) {
241             ctx.lineTo(steps[i].x, steps[i].y);
242         }
243         ctx.stroke();
244 
245     }
246 
247     /**
248      * @description Draws a series of similar lines as a batch. More efficient than drawing them individually.
249      * @param ctx {Object} The canvas context to draw the lines.
250      * @param steps {Object[]} An array of points {x,y}. Each item in the array should contain an array of start and end points.
251      * @param strokeStyle {String} The canvas stroke style.
252      * @param lineWidth {Number} The width of the stroke.
253      * @private
254      */
255 
256     function _drawHaystack(ctx, steps, strokeStyle, lineWidth) {
257 
258         var length = steps.length, i;
259 
260         ctx.strokeStyle = strokeStyle || "#000000";
261         ctx.lineWidth = lineWidth || 1;
262         ctx.beginPath();
263         for (i = 0; i<length; i++) {
264             ctx.moveTo(steps[i][0].x, steps[i][0].y);
265             ctx.lineTo(steps[i][1].x, steps[i][1].y);
266         }
267         ctx.stroke();
268 
269     }
270 
271     function _drawCircle(ctx, center, radius, fillStyle, strokeStyle, lineWidth){
272 
273         ctx.beginPath();
274         ctx.arc(center.x, center.y, radius, 0, Math.PI * 2, false);
275         if (fillStyle) {
276             ctx.fillStyle = fillStyle;
277             ctx.fill();
278         }
279         if (strokeStyle) {
280             ctx.strokeStyle = strokeStyle;
281             ctx.lineWidth = lineWidth || 1;
282             ctx.stroke();
283         }
284 
285     }

292         ctx.fillText(str, pos.x, pos.y, max ? max : null);
293     }

312 VENT.performance = (function() {
313 
314     var _last, _delta, _elapsed;
315 
316     _last = new Date();
317     _delta = 0;
318     _elapsed = 0;
319 
320     function _init() {
321         VENT.renderer.registerRender(_update);
322     }
323 
324     function _update() {
325         var _newDate = new Date();
326         _delta = (_newDate - _last) / 1000;
327         _elapsed += _delta;
328         _last = _newDate;
329     }
330 
331     function _getDelta() {
332         return _delta;
333     }
334 
335     return {
336         init: _init,
337         delta: _delta
338     }
