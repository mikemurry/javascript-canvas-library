 45 /**
 46  * @description Safely augments the JCL namespace to include the specified path.
 47  * @param ns {String} A dot separated namespace. Prefixing with "JCL." is optional.
 48  * @return {Object} Returns the JCL object.
 49  */
 50 
 51 JCL.namespace = function(ns) {
 52 
 53     var parts, parent, i, count;
 54 
 55     parent = JCL;
 56     parts = ns.split(".");
 57 
 58     // Skip Redundant Top Level
 59     if (parts[0] === "JCL") {
 60         parts = parts.slice(1);
 61     }
 62 
 63     count = parts.length;
 64 
 65     for (i=0; i < count; i++) {
 66 
 67         // Create a property if it doesn't exist.
 68         if (typeof parent[parts[i]] === "undefined") {
 69             parent[parts[i]] = {};
 70         }
 71 
 72         // Set parent to new property.
 73         parent = parent[parts[i]];
 74 
 75     }
 76 
 77     return parent;

233 /**
234  * @class The canvas class automatically calculates references to common canvas properties such as the 2d context and sizing information.
235  * @param canvasId {String} The minimum range for the returned integer (inclusive).
236  * @param options {Object} The maximum range for the returned integer (exclusive).
237  * @return {Object}
238  */
239 
240 JCL.Canvas = function Canvas(canvasId, options) {
241 
242     options = options || {};
243     this.options = options;
244 
245     if (typeof canvasId === "object") {
246         this.dom = canvasId[0];
247     } else {
248         this.dom = document.getElementById(canvasId);
249     }
250 
251     if (!this.dom) {
252         JCL.error("Invalid Canvas DOM passed into the constructor.");
253     } else {
254         this.ctx = this.dom.getContext("2d");
255         this.enabled = (this.ctx !== undefined);
256     }
257 
258     if (this.enabled) {
259 
260         // Handle Manual Height/Width Options
261         if (options.width) { this.dom.width = options.width; }
262         if (options.height) { this.dom.height = options.height; }
263 
264         // Handle Full Screen Option
265         if (options.fullscreen) {
266             this.dom.width = window.innerWidth;
267             this.dom.height = window.innerHeight;
268         }
269 
270         // Handle Z-Index Option
271         if (options.zindex) {
272             this.dom.style.zIndex = options.zindex;
273         }
274 
275         this.width = this.dom.width;
276         this.height = this.dom.height;
277 
278         // Handle Margins, Calculate Area
279         options.margins = options.margins || [0,0,0,0];
280         this.area = new JCL.Rectangle(
281             options.margins[3],
282             options.margins[0],
283             this.width - options.margins[1] - options.margins[3],
284             this.height - options.margins[0] - options.margins[2]
285         );
286 
287     } else { JCL.warn("Canvas (ID: '" + canvasId + "') cannot be enabled."); }
288 
289     return this;
290 
291 };

753         return { x: this.x, y: this.y, z: this.z, width: this.width, height: this.height, depth: this.depth };
754     },
755 
756     /**
757      * @description Prints the current coordinates as a string.
758      * @return {String}
759      */
760 
761     toString: function toString() {
762         return "Rectangle( x:" + this.x + ", y:" + this.y + ", z:" + this.z + ", width:" + this.width + ", height:" + this.height + ", depth:" + this.depth + ")";
763     },
764 
765     /**
766      * @description Calculates the center point of the rectangle, or if a point is passed in, adjusts the rectangle's location to match the provided center point.
767      * @return {Object} The resulting point.
768      */
769 
770     center: function center(point) {
771         if (point) {
772             this.set(point.x - (this.width / 2), point.y-(this.height / 2), point.z-(this.depth / 2), this.width, this.height, this.depth);
773         } else {
774             return new JCL.Point(this.x + (this.width / 2), this.y + (this.height / 2), this.z + (this.depth / 2));
775         }
776     }

782 /**
783  * @class The Chart class makes it easier to map data points based on values instead of physical canvas coordinates.
784  * @param area {Object} The rectangle or canvas that specifies the boundaries of the chart.
785  * @param options {Object}
786  * @return {Object}
787  */
788 
789 JCL.Chart = function Chart(area, options) {
790 
791     options = options || {};
792     this.options = options;
793 
794     if (area) {
795 
796         // Initialize Area
797         if (area instanceof JCL.Canvas) {
798             this.area = new JCL.Rectangle(0, 0, area.width, area.height);
799         } else if (area instanceof JCL.Rectangle) {
800             this.area = area;
801         }
802 
803         // Initialize X and Y Axis
804         options.xaxis = options.xaxis || {};
805         options.yaxis = options.yaxis || {};
806         this.xaxis = {
807             min: options.xaxis.min || 0,
808             max: options.xaxis.max || 100,
809             interval: options.xaxis.interval || 10,
810             type: options.xaxis.type || null
811         };
812         this.yaxis = {
813             min: options.yaxis.min || 0,
814             max: options.yaxis.max || 100,
815             interval: options.yaxis.interval || 10
816         };
817 
818     } else {
819         JCL.error("Cannot create a Chart without a Canvas or Rectangle specified as the area.")
820     }
821 
822     return this;
823 
824 };
825 
826 JCL.Chart.prototype = {
827 
828     plot: function plot(x, y) {
829 
830         var intervalWidth;
831 
832         if (this.xaxis.type === "column") {
833             intervalWidth = this.area.width / (this.xaxis.max - this.xaxis.min + 1);
834             return new JCL.Point(
835                 this.area.x + Math.floor(((x - this.xaxis.min) * intervalWidth) + (intervalWidth/2)),
836                 this.area.y + this.area.height - Math.floor((((y - this.yaxis.min)/(this.yaxis.max - this.yaxis.min)) * this.area.height))
837             );
838         }
839 
840         return new JCL.Point(
841             Math.floor(this.area.x + (((x - this.xaxis.min) / (this.xaxis.max - this.xaxis.min)) * this.area.width)),
842             Math.floor(this.area.y + this.area.height - (((y - this.yaxis.min)/(this.yaxis.max - this.yaxis.min)) * this.area.height))
843         );
844 
845     }
