 45 /**
 46  * @description Safely augments the JCL namespace to include the specified path.
 47  * @param ns {String} A dot separated namespace. Prefixing with "JCL." is optional.
 48  * @return {Object} Returns the JCL object.
 49  */
 50 
 51 JCL.namespace = function(ns) {
 52 
 53     var parts, parent, i, count;
 54 
 55     parent = JCL;
 56     parts = ns.split(".");
 57 
 58     // Skip Redundant Top Level
 59     if (parts[0] === "JCL") {
 60         parts = parts.slice(1);
 61     }
 62 
 63     count = parts.length;
 64 
 65     for (i=0; i < count; i++) {
 66 
 67         // Create a property if it doesn't exist.
 68         if (typeof parent[parts[i]] === "undefined") {
 69             parent[parts[i]] = {};
 70         }
 71 
 72         // Set parent to new property.
 73         parent = parent[parts[i]];
 74 
 75     }
 76 
 77     return parent;

206 /**
207  * @class The canvas class automatically calculates references to common canvas properties such as the 2d context and sizing information.
208  * @param canvasId {String} The minimum range for the returned integer (inclusive).
209  * @param options {Object} The maximum range for the returned integer (exclusive).
210  * @return {Object}
211  */
212 
213 JCL.Canvas = function Canvas(canvasId, options) {
214 
215     options = options || {};
216     this.options = options;
217 
218     if (typeof canvasId === "object") {
219         this.dom = canvasId[0];
220     } else {
221         this.dom = document.getElementById(canvasId);
222     }
223 
224     if (!this.dom) {
225         JCL.error("Invalid Canvas DOM passed into the constructor.");
226     } else {
227         this.ctx = this.dom.getContext("2d");
228         this.enabled = (this.ctx !== undefined);
229     }
230 
231     if (this.enabled) {
232 
233         // Handle Manual Height/Width Options
234         if (options.width) { this.dom.width = options.width; }
235         if (options.height) { this.dom.height = options.height; }
236 
237         // Handle Full Screen Option
238         if (options.fullscreen) {
239             this.dom.width = window.innerWidth;
240             this.dom.height = window.innerHeight;
241         }
242 
243         // Handle Z-Index Option
244         if (options.zindex) {
245             this.dom.style.zIndex = options.zindex;
246         }
247 
248         this.width = this.dom.width;
249         this.height = this.dom.height;
250 
251         // Handle Margins, Calculate Area
252         options.margins = options.margins || [0,0,0,0];
253         this.area = {
254             x: options.margins[3],
255             y: options.margins[0],
256             width: this.width - options.margins[1] - options.margins[3],
257             height: this.height - options.margins[0] - options.margins[2]
258         };
259 
260     } else { JCL.warn("Canvas (ID: '" + canvasId + "') cannot be enabled."); }
261 
262     return this;
263 
264 };

480 JCL.performance = (function() {
481 
482     var _isInit, _last, _delta, _fps, _elapsed;
483 
484     _isInit = false;
485     _last = new Date();
486     _delta = 0;
487     _elapsed = 0;
488     _fps = 0;
489 
490     function _init() {
491         if (!_isInit) {
492             JCL.renderer.registerRender(_update);
493             _isInit = true;
494         }
495     }
496 
497     function _update() {
498         var _newDate = new Date();
499         _delta = (_newDate - _last) / 1000;
500         _elapsed += _delta;
501         _last = _newDate;
502         _fps = Math.floor(1/_delta);
503     }
504 
505     _init();
506 
507     return {
508 
509         /**
510          * @property {Number} delta The number of seconds elapsed since the last rendered frame.
511          */
512 
513         delta: _delta,
514 
515         /**
516          * @property {Number} elapsed The number of seconds elapsed since page load.
517          */
518 
519         elapsed: _elapsed,

523          */

534  * @class The point class represents a point in 2d (x,y) or 3d (x,y,z) space.

541 JCL.Point = function Point(x, y, z) {
542     return this.set(x,y,z);
543 };
544 
545 JCL.Point.prototype = {
546 
547     constructor: "Point",
548 
549     /**
550      * @description Calculates the angle between two points.
551      * @param x {Number} The x coordinate.
552      * @param y {Number} The y coordinate.
553      * @param z {Number} The z coordinate.
554      * @return {Object} The resulting point object.
555      */
556 
557     set: function set(x,y,z) {
558         this.x = x || 0;
559         this.y = y || 0;
560         this.z = z || 0;
561         return this;
562     },
563 
564     /**
565      * @description Returns a simplified object representing the point's coordinates.
566      * @return {Object}
567      */
568 
569     get: function get() {
570         return { x: this.x, y: this.y, z: this.z };
571     },
572 
573     /**
574      * @description Prints the current coordinates as a string.
575      * @return {String}
576      */
577 
578     toString: function toString() {
579         return "Point( x:" + this.x + ", y:" + this.y + ", z:" + this.z + ")";
580     },
581 
582     /**
583      * @description Calculates the distance between two points.
584      * @param a {Object} The first point.
585      * @return {Number|Undefined} The resulting length.
586      */
587 
588     distance: function distance(a) {
589         if (a.constructor === "Point") { return Math.sqrt(Math.pow(a.x- this.x, 2) + Math.pow(a.y- this.y,2)); }
590         else { JCL.warn("Cannot calculate distance of a non-point."); }
591         return undefined;
592     },
593 
594     /**
595      * @description Calculates the angle between two points.
596      * @param a {Object} The first point.
597      * @return {Number|Undefined} The resulting angle in radians.
598      */
599 
600     angle: function angle(a) {
601         if (a.constructor === "Point") { return Math.atan2(a.y- this.y, a.x - this.x); }
602         else { JCL.warn("Cannot calculate angle of a non-point."); }
603         return undefined;
604     },
605 
606     /**
607      * @description Calculates a point {x,y} at the specified angle and distance from the base point.
608      * @param angle {Number} The number of degrees to rotate the resulting point around the center (0 is pointing right).
609      * @param distance {Number} The number of pixels between the center and the resulting point.
610      * @return {Object|Undefined} The resulting point expressed as an object with 'x' and 'y' properties.
611      */
612 
613     tangent: function tangent(angle, distance) {
614 
615         var radians = JCL.utilities.radians;
616 
617         if (angle !== null && distance !== null) {
618             return new JCL.Point((distance * Math.cos(radians(angle))) + this.x, (distance * Math.sin(radians(angle))) + this.y);
619         } else { JCL.warn("Could not calculate tangent. Missing required data. (Angle: " + angle + ", Distance: " + distance + ")"); }
620         return undefined;
621 
622     },
623 
624     /**
625      * @description Interpolates between two points {x,y}.
626      * @param a {Object} The first point {x,y}.
627      * @param ratio {Number} The amount to interpolate between the two points. Between 0 and 1.
628      * @return {Object|Undefined} The interpolated point {x,y}.
629      */
630 
631     lerp: function lerp(a,ratio) {
632         if (!ratio || ratio < 0) { ratio = 0; }
633         else if (ratio > 1) { ratio = 1; }
634         if (this && a && ratio !== null) {
635             if (a.constructor === "Point") {
636                 return { x: this.x + ((a.x - this.x) * ratio), y: this.y + ((a.y - this.y) * ratio) };
637             } else { JCL.warn("Could not interpolate point. Invalid point."); }
638         } else { JCL.warn("Could not interpolate point. Missing required data."); }
639         return undefined;
640     }
