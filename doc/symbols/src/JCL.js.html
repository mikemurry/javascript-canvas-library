 25 /**
 26  * @description Safely augments the JCL namespace to include the specified path.
 27  * @param ns {String} A dot separated namespace. Prefixing with "JCL." is optional.
 28  * @return {Object} Returns the JCL object.
 29  */
 30 
 31 JCL.namespace = function(ns) {
 32 
 33     var parts, parent, i, count;
 34 
 35     parent = JCL;
 36     parts = ns.split(".");
 37 
 38     // Skip Redundant Top Level
 39     if (parts[0] === "JCL") {
 40         parts = parts.slice(1);
 41     }
 42 
 43     count = parts.length;
 44 
 45     for (i=0; i < count; i++) {
 46 
 47         // Create a property if it doesn't exist.
 48         if (typeof parent[parts[i]] === "undefined") {
 49             parent[parts[i]] = {};
 50         }
 51 
 52         // Set parent to new property.
 53         parent = parent[parts[i]];
 54 
 55     }
 56 
 57     return parent;
 58

118         return window.requestAnimationFrame ||

128     _list = [];

130 
131     function _loop() {
132         if (_len > 0) {
133             for (var i = 0; i < _len; i++) { _list[i](); }

135         }
136     }

138     function _register(fn) {
139         var len = _list.push(fn);
140         _len = len;
141         if (_len == 1) { _loop(); }

177 /**
178  * @class The canvas class automatically calculates references to common canvas properties such as the 2d context and sizing information.
179  * @param canvasId {String} The minimum range for the returned integer (inclusive).
180  * @param options {Object} The maximum range for the returned integer (exclusive).
181  * @return {Object}
182  */
183 
184 JCL.Canvas = function Canvas(canvasId, options) {
185 
186     this.dom = document.getElementById(canvasId);
187     this.ctx = this.dom.getContext("2d");
188     this.enabled = (this.ctx !== undefined);
189 
190     if (this.enabled) {
191 
192         // Handle Manual Height/Width Options
193         if (options.width) { this.dom.width = options.width; }
194         if (options.height) { this.dom.height = options.height; }
195 
196         // Handle Full Screen Option
197         if (options.fullscreen) {
198             this.dom.width = window.innerWidth;
199             this.dom.height = window.innerHeight;
200         }
201 
202         // Handle Z-Index Option
203         if (options.zindex) {
204             this.dom.style.zIndex = options.zindex;
205         }
206 
207         this.width = this.dom.width;
208         this.height = this.dom.height;
209 
210     } else { JCL.warn("Canvas (ID: '" + canvasId + "') cannot be enabled."); }
211 
212     return this;
213 
214 };

355 JCL.performance = (function() {
356 
357     var _last, _delta, _fps, _elapsed, _dom;
358 
359     _last = new Date();
360     _delta = 0;
361     _elapsed = 0;
362     _fps = 0;
363 
364     function _init(domId) {
365         if (domId) { _dom = document.getElementById(domId); }
366         JCL.renderer.registerRender(_update);
367     }
368 
369     function _update() {
370         var _newDate = new Date();
371         _delta = (_newDate - _last) / 1000;
372         _elapsed += _delta;
373         _last = _newDate;
374         _fps = Math.floor(1/_delta);
375         if (_dom) {
376             _dom.innerHTML = _fps;
377         }
378     }
379 
380     function _getDelta() {
381         return _delta;
382     }
383 
384     function _getElapsed() {
385         return _elapsed;
386     }

388     return {
389 
390         /**
391          * @function
392          * @description Initializes the performance functions.
393          * @param domId {String} If a ID string is passed to a DOM element, that element's text will be updated with the current frames per second (FPS).
394          * @return {Number}
395          */
396 
397         init: _init,
398 
399         /**
400          * @function
401          * @description Returns the latest delta (elapsed time since last rendered frame, in seconds).
402          * @return {Number}
403          */

405         delta: _getDelta,

428     constructor: "Point",
429 
430     set: function set(x,y,z) {
431         this.x = x || 0;
432         this.y = y || 0;
433         this.z = z || 0;
434     },
435 
436     get: function get() {
437         return { x: this.x, y: this.y, z: this.z };
438     },
439 
440     /**
441      * @description Calculates the distance between two points.
442      * @param a {Object} The first point.
443      * @return {Number|Undefined} The resulting length.
444      */
445 
446     distance: function distance(a) {
447         if (a.constructor === "Point") { return Math.sqrt(Math.pow(a.x- this.x, 2) + Math.pow(a.y- this.y,2)); }
448         else { JCL.warn("Cannot calculate distance of a non-point."); }
449         return undefined;
450     },
451 
452     /**
453      * @description Calculates the angle between two points.
454      * @param a {Object} The first point.
455      * @return {Number|Undefined} The resulting angle in radians.
456      */
457 
458     angle: function angle(a) {
459         if (a.constructor === "Point") { return Math.atan2(a.y- this.y, a.x - this.x); }
460         else { JCL.warn("Cannot calculate angle of a non-point."); }
461         return undefined;
462     },
463 
464     /**
465      * @description Calculates a point {x,y} at the specified angle and distance from the base point.
466      * @param angle {Number} The number of degrees to rotate the resulting point around the center (0 is pointing right).
467      * @param distance {Number} The number of pixels between the center and the resulting point.
468      * @return {Object|Undefined} The resulting point expressed as an object with 'x' and 'y' properties.
469      */
470 
471     tangent: function tangent(angle, distance) {
472 
473         var radians = JCL.utilities.radians;
474 
475         if (angle !== null && distance !== null) {
476             return new JCL.Point((distance * Math.cos(radians(angle))) + this.x, (distance * Math.sin(radians(angle))) + this.y);
477         } else { JCL.warn("Could not calculate tangent. Missing required data. (Angle: " + angle + ", Distance: " + distance + ")"); }
478         return undefined;
479 
480     },
481 
482     /**
483      * @description Interpolates between two points {x,y}.
484      * @param a {Object} The first point {x,y}.
485      * @param ratio {Number} The amount to interpolate between the two points. Between 0 and 1.
486      * @return {Object|Undefined} The interpolated point {x,y}.
487      */
488 
489     lerp: function lerp(a,ratio) {
490         if (!ratio || ratio < 0) { ratio = 0; }
491         else if (ratio > 1) { ratio = 1; }
492         if (this && a && ratio !== null) {
493             if (a.constructor === "Point") {
494                 return { x: this.x + ((a.x - this.x) * ratio), y: this.y + ((a.y - this.y) * ratio) };
495             } else { JCL.warn("Could not interpolate point. Invalid point."); }
496         } else { JCL.warn("Could not interpolate point. Missing required data."); }
497         return undefined;
498     }
